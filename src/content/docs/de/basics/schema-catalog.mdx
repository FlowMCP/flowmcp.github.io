export const SCHEMAS_URL = 'https://raw.githubusercontent.com/flowmcp/flowmcp-schemas-public/main/schemas.json'
export const GITHUB_BASE = 'https://github.com/flowmcp/flowmcp-schemas-public/tree/main/schemas'

export const SHOW_PROVIDERS = ['brightsky', 'transportrest', 'dbris', 'nextbike', 'nominatim', 'overpass', 'vlocker', 'infravelo', 'datetimeutils']

export const fetchSchemas = async () => {
    try {
        const res = await fetch(SCHEMAS_URL)
        return await res.json()
    } catch (e) {
        return { providers: [], totalProviders: 0, totalTools: 0, totalSchemas: 0 }
    }
}

export const data = await fetchSchemas()
export const filtered = (data.providers || []).filter(p => SHOW_PROVIDERS.includes(p.folder))
